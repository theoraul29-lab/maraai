const express = require("express");
const path = require("path");
const fs = require("fs");

const app = express();
const PORT = process.env.PORT || 3001;

// Path to frontend build
const DIST_DIR = path.join(__dirname, "../../frontend/dist");

// Serve static files
app.use(express.static(DIST_DIR));

// Catch-all fallback: serve index.html for SPA routes
app.get("*", (req, res) => {
  const indexFile = path.join(DIST_DIR, "index.html");
  if (fs.existsSync(indexFile)) {
    res.sendFile(indexFile);
  } else {
    res.send(`
      <!DOCTYPE html>
      <html>
        <head>
          <meta charset="UTF-8" />
          <title>MaraAI Fallback</title>
        </head>
        <body>
          <h1>MaraAI is running fallback mode!</h1>
          <p>Frontend not built yet, please run <code>npm run build</code> in frontend/</p>
        </body>
      </html>
    `);
  }
});

// Generic error handler
app.use((err, req, res, next) => {
  console.error("Server error:", err);
  res.status(500).send("Internal Server Error (handled by wrapper)");
});

app.listen(PORT, () => {
  console.log(`MaraAI wrapper server running on http://localhost:${PORT}`);
  console.log("All errors will be caught, and static fallback is enabled.");
});
