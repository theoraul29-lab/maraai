<script>
const feedDiv = document.getElementById('feed');
const audio = document.getElementById('maraAudio');

async function loadFeed(){
    const res = await fetch('http://localhost:3000/api/feed');
    const feed = await res.json();
    feedDiv.innerHTML='';
    feed.forEach((r,index)=>{
        const div=document.createElement('div');
        div.className='reel';
        div.innerHTML=`
            <video src="${r.url}" muted playsinline></video>
            <button onclick="openFullscreen(${index})">Fullscreen + Sound</button>
        `;
        feedDiv.appendChild(div);
    });
    setupAutoplay();
}

async function askMara(){
    const text=document.getElementById('questionText').value;
    const voice=document.getElementById('voiceSelect').value;
    const userId='USER123'; // ID demo
    const res=await fetch('http://localhost:3000/api/mara-tts',{
        method:'POST',
        headers:{'Content-Type':'application/json'},
        body:JSON.stringify({userId,text,voice})
    });
    const data=await res.json();
    if(data.error) alert(data.error);
    else {
        audio.src=data.audioURL;
        audio.play();
    }
}

// Scroll autoplay video
function setupAutoplay(){
    const videos=document.querySelectorAll('video');
    feedDiv.addEventListener('scroll',()=>{
        videos.forEach(v=>{
            const rect=v.getBoundingClientRect();
            if(rect.top>=0 && rect.bottom<=window.innerHeight)v.play();
            else v.pause();
        });
    });
}

// Fullscreen video
function openFullscreen(index){
    const video=document.querySelectorAll('video')[index];
    if(video.requestFullscreen) video.requestFullscreen();
    video.muted=false;
    video.play();
}

loadFeed();
