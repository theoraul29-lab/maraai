<script>
// Storage for writers
let writers = [];
let isSubscribed = false;
let subscriptionType = null;

// Subscription system
function subscribe(type){
    subscriptionType = type;
    isSubscribed = true;
    alert(`Subscription activated: ${type === 'monthly' ? '1 month' : '1 year'} full access!`);
}

// Create writer page
function createPage(){
    const output = document.getElementById('output');
    output.innerHTML = `
        <h2>Create Your Writer Page</h2>
        <input type="text" id="writerName" placeholder="Your pen name">
        <input type="text" id="pageTitle" placeholder="Page / Book Title">
        <textarea id="contentArea" rows="8" placeholder="Write your story, poem, or chapter here..."></textarea>
        <input type="file" id="uploadImage" accept="image/*">
        <button onclick="insertImage()">Insert Image</button>
        <button onclick="previewWriterPage()">Preview Page</button>
    `;
}

// Insert image
function insertImage(){
    const fileInput = document.getElementById('uploadImage');
    if(fileInput.files && fileInput.files[0]){
        const reader = new FileReader();
        reader.onload = function(e){
            const img = document.createElement('img');
            img.src = e.target.result;
            document.getElementById('output').appendChild(img);
        }
        reader.readAsDataURL(fileInput.files[0]);
    } else {
        alert("Please select an image first.");
    }
}

// Preview writer page
function previewWriterPage(){
    const name = document.getElementById('writerName').value || "Anonymous";
    const title = document.getElementById('pageTitle').value || "Untitled";
    const content = document.getElementById('contentArea').value || "(No content)";
    
    // Mara AI live suggestions demo
    const suggestion = isSubscribed 
        ? "Mara AI Suggestion: Make your first sentence captivating and engaging!"
        : "Subscribe to unlock full Mara AI feedback and premium features.";

    // Save writer page
    writers.push({name, title, content});

    const output = document.getElementById('output');
    output.innerHTML = `
        <div class="writerCard">
            <h2>${title}</h2>
            <h4>by ${name}</h4>
            <p>${content}</p>
            <p style="color:#0ff;"><em>${suggestion}</em></p>
            <button onclick="editWriterPage()">Edit Page</button>
            <button onclick="publishPage('${name}','${title}','${content}')">Publish Online</button>
        </div>
    `;
}

// Edit page
function editWriterPage(){
    const lastPage = writers[writers.length-1];
    const output = document.getElementById('output');
    output.innerHTML = `
        <h2>Edit Your Writer Page</h2>
        <input type="text" id="writerName" value="${lastPage.name}" placeholder="Your pen name">
        <input type="text" id="pageTitle" value="${lastPage.title}" placeholder="Page / Book Title">
        <textarea id="contentArea" rows="8">${lastPage.content}</textarea>
        <input type="file" id="uploadImage" accept="image/*">
        <button onclick="insertImage()">Insert Image</button>
        <button onclick="previewWriterPage()">Preview Page</button>
    `;
}

// Publish page online (demo)
function publishPage(name,title,content){
    if(!isSubscribed){
        return alert("You need a premium subscription to publish online!");
    }

    // Simulated online publication: add to output list
    const output = document.getElementById('output');
    output.innerHTML += `
        <div class="writerCard">
            <h2>${title}</h2>
            <h4>by ${name}</h4>
            <p>${content}</p>
            <p style="color:#0f0;"><em>Published online! Visible to readers.</em></p>
        </div>
    `;
    alert(`Your page "${title}" is now published online and visible to readers!`);
}
</script>
