    <input type="text" id="user-input" placeholder="Type your message..." autofocus>
    <button id="send-btn">Send</button>
    <select id="voice-select">
      <option value="female">Female</option>
      <option value="male">Male</option>
      <option value="robot">Robot</option>
    </select>
    <button id="voice-btn">Speak</button>

<script>
  const messagesEl = document.getElementById('messages');
  const userInput = document.getElementById('user-input');
  const sendBtn = document.getElementById('send-btn');
  const voiceBtn = document.getElementById('voice-btn');
  const voiceSelect = document.getElementById('voice-select');

  let colors = ['#111', '#1a1a1a', '#222', '#2a2a2a', '#333'];
  let colorIndex = 0;

  let freeVoiceLimit = 5; // max 5 voice responses for free
  let isPremium = false;   // change to true for unlimited voice

  // Load saved conversation
  let conversation = JSON.parse(localStorage.getItem('mara_convo') || '[]');
  conversation.forEach(msg => addMessage(msg.text, msg.sender));

  function addMessage(text, sender) {
    const msg = document.createElement('div');
    msg.classList.add('message', sender);
    msg.textContent = text;
    messagesEl.appendChild(msg);
    messagesEl.scrollTop = messagesEl.scrollHeight;

    // Save message
    conversation.push({text, sender});
    localStorage.setItem('mara_convo', JSON.stringify(conversation));

    // Change ambient background slowly
    document.body.style.background = colors[colorIndex % colors.length];
    colorIndex++;
  }

  function maraReply(userText) {
    let reply = "Mara AI: I received your message: " + userText;

    // Example smart responses
    if(userText.toLowerCase().includes('trading')) {
      reply = "Mara AI: Trading is buying/selling assets to profit. I can explain strategies!";
    } else if(userText.toLowerCase().includes('story')) {
      reply = "Mara AI: Let's create your story! What's your idea?";
    }

    setTimeout(() => addMessage(reply, 'mara'), 800);
  }

  function sendMessage() {
    const text = userInput.value.trim();
    if(!text) return;
    addMessage(text, 'user');
    userInput.value = '';
    maraReply(text);
  }

  sendBtn.addEventListener('click', sendMessage);
  userInput.addEventListener('keydown', e => { if(e.key==='Enter') sendMessage(); });

  // Voice functionality
  voiceBtn.addEventListener('click', () => {
    if(!isPremium && freeVoiceLimit <= 0){
      alert("Upgrade to Premium for unlimited voice responses!");
      return;
    }
    const lastMsg = conversation[conversation.length-1];
    if(!lastMsg) return;

    const utterance = new SpeechSynthesisUtterance(lastMsg.text);
    switch(voiceSelect.value){
      case 'female': utterance.voice = speechSynthesis.getVoices().find(v=>v.name.includes('Female')) || null; break;
      case 'male': utterance.voice = speechSynthesis.getVoices().find(v=>v.name.includes('Male')) || null; break;
      case 'robot': utterance.pitch=0.3; utterance.rate=1.1; break;
    }
    speechSynthesis.speak(utterance);

    if(!isPremium) freeVoiceLimit--;
  });
</script>
