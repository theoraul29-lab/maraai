<script>
// Mara Activation
const maraBtn = document.getElementById('maraBtn');
const tradingBtn = document.getElementById('tradingBtn');
const output = document.getElementById('output');
const maraInput = document.getElementById('maraInput');

let maraActive = false;
maraBtn.addEventListener('click', ()=>{
    maraActive = true;
    output.innerHTML = "<p>Mara is now active. Ask me anything or press the Trading Tutorial button!</p>";
    maraInput.focus();
});

// Mara Trading Tutorial Function
tradingBtn.addEventListener('click', ()=>{
    if(!maraActive){
        alert("Activate Mara first!");
        return;
    }

    // Example explanation like a 30-year-experienced trader
    output.innerHTML = `
    <h3>Welcome to Trading 101 - Mara AI Explains</h3>
    <p><strong>1. What is Trading?</strong> Trading is the process of buying and selling assets like cryptocurrencies, stocks, or NFTs to make profits. A seasoned trader studies the market, understands patterns, and manages risk carefully.</p>
    <p><strong>2. Using Charts:</strong> Charts show the price history of an asset. Candlestick charts reveal trends, support & resistance levels, and help you decide entry and exit points.</p>
    <p><strong>3. Trading Types:</strong> Spot trading = buy/sell actual assets. Futures = contracts predicting price changes. DeFi = decentralized finance protocols allow lending, staking, yield farming. NFTs = unique digital assets with value based on demand.</p>
    <p><strong>4. Advanced Tips:</strong> Always manage risk: use stop-loss, take-profit, diversify. Analyze volume, trends, and indicators like RSI, MACD. A trader with 30 years of experience emphasizes patience, discipline, and constant learning.</p>
    <p><strong>5. Interactive Learning:</strong> Ask Mara about charts, specific strategies, or DeFi/NFT examples. She can explain step-by-step, just like a professional mentor.</p>
    `;
});

// Optional: Respond to user input
maraInput.addEventListener('keypress', (e)=>{
    if(e.key === 'Enter' && maraActive){
        const question = maraInput.value;
        maraInput.value = '';
        output.innerHTML += `<p><strong>You asked:</strong> ${question}</p>`;
        // Example response logic
        output.innerHTML += `<p><strong>Mara answers:</strong> As a seasoned trader, here's my advice: Focus on understanding charts first. Identify trends and volume before making trades. For DeFi, research smart contracts. For NFTs, understand rarity and demand.</p>`;
        output.scrollTop = output.scrollHeight;
    }
});
</script>
