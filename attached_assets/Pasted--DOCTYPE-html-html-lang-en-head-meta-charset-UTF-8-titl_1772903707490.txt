<script>
let isSubscribed = false;
let subscriptionType = null;

function subscribe(type){
    subscriptionType = type;
    isSubscribed = true;
    alert(`Subscription activated: ${type === 'monthly' ? '1 month' : '1 year'} full access!`);
    loadStrategies();
}

// Generate demo strategies
const strategies = [];
for(let i=1;i<=50;i++){
    strategies.push({
        title: `Strategy ${i}: Simple Trading Technique`,
        spot: `Spot explanation for Strategy ${i}: buy/sell rules, entry, exit points.`,
        futures: `Futures explanation for Strategy ${i}: leverage tips, risk management, TP/SL.`,
        image: `https://upload.wikimedia.org/wikipedia/commons/thumb/3/34/Stock_chart_example.png/320px-Stock_chart_example.png`
    });
}

// Load strategies in UI
function loadStrategies(){
    const academy = document.getElementById('academy');
    academy.innerHTML = '';
    strategies.forEach(s=>{
        const div = document.createElement('div');
        div.className = 'strategyCard';
        if(isSubscribed){
            div.innerHTML = `
                <h3>${s.title}</h3>
                <p><strong>Spot:</strong> ${s.spot}</p>
                <p><strong>Futures:</strong> ${s.futures}</p>
                <img src="${s.image}" alt="${s.title}" style="max-width:100%; border-radius:6px;">
            `;
        } else {
            div.innerHTML = `
                <h3>${s.title}</h3>
                <p>Subscribe to see full Spot & Futures details and examples.</p>
            `;
        }
        academy.appendChild(div);
    });
}

// Initial load (preview only)
strategies.forEach(s=>{
    const academy = document.getElementById('academy');

    div.className = 'strategyCard';
    div.innerHTML = `<h3>${s.title}</h3><p>Subscribe to unlock full details.</p>`;
    academy.appendChild(div);
});
