<script>
const maraBtn = document.getElementById('maraBtn');
const maraInput = document.getElementById('maraInput');
const reelsContainer = document.getElementById('reelsContainer');

// List of YouTube Shorts/Reels URLs (replace with real video IDs)
const youtubeShorts = [
  "SHORT_VIDEO_ID1",
  "SHORT_VIDEO_ID2",
  "SHORT_VIDEO_ID3"
];

// Function for Mara to “post” a new reel
function maraPostReel(videoId){
    const iframe = document.createElement("iframe");
    iframe.className = "reel";
    iframe.src = `https://www.youtube.com/embed/${videoId}?autoplay=1&mute=1&controls=1&loop=1`;
    iframe.allow = "accelerometer; autoplay; clipboard-write; encrypted-media; gyroscope; picture-in-picture";
    iframe.allowFullscreen = true;
    reelsContainer.prepend(iframe); // newest reels appear on top
}

// When Mara is activated
maraBtn.addEventListener('click', ()=>{
    alert("Mara is now active! She can post reels.");
    maraInput.focus();
    
    // Example: Mara posts all reels one by one
    youtubeShorts.forEach(id => maraPostReel(id));
});

// Optional: Post reel when user types something
maraInput.addEventListener('keypress', (e)=>{
    if(e.key === 'Enter'){
        const text = maraInput.value;
        maraInput.value = '';
        // Example logic: Mara picks a random reel to post based on text
        const randomId = youtubeShorts[Math.floor(Math.random()*youtubeShorts.length)];
        maraPostReel(randomId);
        console.log(`Mara posted a reel in response to: "${text}"`);
    }
});
</script>
