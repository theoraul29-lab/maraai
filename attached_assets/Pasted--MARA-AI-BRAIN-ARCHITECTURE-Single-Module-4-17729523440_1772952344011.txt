/**
 * MARA AI BRAIN ARCHITECTURE (Single Module)
 * ------------------------------------------
 * 4 AI Agents:
 *  - ResearchAgent: scans web trends and knowledge
 *  - ProductAgent: proposes features & UI improvements
 *  - DeveloperAgent: proposes technical tasks/patches (no auto-apply)
 *  - GrowthAgent: suggests user growth strategies
 *
 * Stack: Node.js, Express, MongoDB, OpenAI, Axios
 * Endpoint: POST /api/mara-brain/run
 */

const express = require("express");
const mongoose = require("mongoose");
const axios = require("axios");
const OpenAI = require("openai");

const app = express();
app.use(express.json());

// --------- CONFIG ---------
const PORT = process.env.PORT || 5050;
const MONGO_URI = process.env.MONGO_URI || "mongodb://127.0.0.1:27017/mara_ai";
const OPENAI_KEY = process.env.OPENAI_KEY;

// --------- DB ---------
mongoose.connect(MONGO_URI);

// Logs of each brain cycle
const BrainLogSchema = new mongoose.Schema({
  research: Object,
  productIdeas: String,
  devTasks: String,
  growthIdeas: String,
  createdAt: { type: Date, default: Date.now }
});
const BrainLog = mongoose.model("BrainLog", BrainLogSchema);

// --------- OPENAI CLIENT ---------
const openai = new OpenAI({ apiKey: OPENAI_KEY });

// --------- AGENT 1: RESEARCH ---------
async function ResearchAgent() {
  const query = "AI education platforms trends AI chat apps features";

  try {
    const res = await axios.get(
      `https://api.duckduckgo.com/?q=${encodeURIComponent(query)}&format=json`
    );
    return res.data.RelatedTopics || [];
  } catch {
    return [];
  }
}

// --------- AGENT 2: PRODUCT ---------
async function ProductAgent(researchData) {
  const prompt = `
You are Mara AI Product Architect.

Based on these research trends, suggest:
1) new features
2) UI improvements
3) product innovations

Research data:
${JSON.stringify(researchData).slice(0, 2000)}
`;

  const completion = await openai.chat.completions.create({
    model: "gpt-4o-mini",
    messages: [
      { role: "system", content: "You design AI platforms." },
      { role: "user", content: prompt }
    ]
  });

  return completion.choices[0].message.content;
}

// --------- AGENT 3: DEVELOPER ---------
async function DeveloperAgent(productIdeas) {
  const prompt = `
You are a senior software engineer with 30 years of experience.

Turn these product ideas into:
- backend tasks
- frontend tasks
- database changes
- possible code patches

Product ideas:
${productIdeas}
`;

  const completion = await openai.chat.completions.create({
    model: "gpt-4o-mini",
    messages: [
      { role: "system", content: "You plan software architecture." },
      { role: "user", content: prompt }
    ]
  });

  return completion.choices[0].message.content;
}

// --------- AGENT 4: GROWTH ---------
async function GrowthAgent(productIdeas) {
  const prompt = `
You are a startup growth strategist.

Based on these product ideas suggest:
- marketing strategies
- user growth tactics
- community ideas
- monetization improvements

Ideas:
${productIdeas}
`;

  const completion = await openai.chat.completions.create({
    model: "gpt-4o-mini",
    messages: [
      { role: "system", content: "You grow tech startups." },
      { role: "user", content: prompt }
    ]
  });

  return completion.choices[0].message.content;
}

// --------- BRAIN ORCHESTRATOR ---------
async function MaraBrainCycle() {
  const research = await ResearchAgent();
  const productIdeas = await ProductAgent(research);
  const devTasks = await DeveloperAgent(productIdeas);
  const growthIdeas = await GrowthAgent(productIdeas);

  const log = new BrainLog({
    research,
    productIdeas,
    devTasks,
    growthIdeas
  });

  await log.save();

  return log;
}

// --------- API ---------
app.post("/api/mara-brain/run", async (req, res) => {
  try {
    const result = await MaraBrainCycle();
    res.json({
      message: "Mara AI Brain cycle executed",
      result
    });
  } catch (e) {
    res.status(500).json({ error: e.message });
  }
});

// --------- SERVER ---------
app.listen(PORT, () => {
  console.log(`Mara AI Brain running on port ${PORT}`);
});
