<script>

/* paleta culori ambientale calde */

const ambientColors=[
"#1a1a1a",
"#2b1a12",
"#3a1e12",
"#332211",
"#2a1a1a",
"#3a2418"
]

function changeAmbient(){
let color=ambientColors[Math.floor(Math.random()*ambientColors.length)]
document.body.style.background=color
}

/* chat */

const chat=document.getElementById("chatWindow")

function addMessage(text,type){

let msg=document.createElement("div")
msg.className="message "+type
msg.innerText=text

chat.appendChild(msg)
chat.scrollTop=chat.scrollHeight

}

/* simulare raspuns Mara */

function maraReply(userText){

changeAmbient()

setTimeout(()=>{

let response="Am primit mesajul: "+userText+"."

addMessage(response,"mara")

changeAmbient()

},1200)

}

/* trimitere mesaj */

function sendMessage(){

let input=document.getElementById("textInput")
let text=input.value.trim()

if(!text)return

addMessage(text,"user")

input.value=""

changeAmbient()

maraReply(text)

}

/* enter key */

document.getElementById("textInput").addEventListener("keypress",function(e){

if(e.key==="Enter"){
sendMessage()
}

})

</script>
