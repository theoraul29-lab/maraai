<script>
// Mara activation
let voiceMode = false;
document.getElementById('voiceBtn').addEventListener('click', ()=>{
    voiceMode = !voiceMode;
    document.getElementById('voiceBtn').innerText = "Voice Mode: " + (voiceMode ? "ON" : "OFF");
});

// Tutorial content
const tutorialContent = [
    {
        title: "What is Trading?",
        text: "Trading is the process of buying and selling financial assets such as cryptocurrencies, stocks, commodities, and NFTs to earn profits. A professional trader studies market trends, charts, and risk management techniques.",

    },
    {
        title: "Spot vs Futures",
        text: "Spot trading involves buying or selling the actual asset. Futures trading involves contracts predicting future prices. Futures allow leverage but carry higher risk.",
        img: "https://upload.wikimedia.org/wikipedia/commons/thumb/7/7e/Futures_chart_example.png/640px-Futures_chart_example.png"
    },
    {
        title: "DeFi & NFTs",
        text: "DeFi (Decentralized Finance) allows lending, staking, and yield farming. NFTs are unique digital assets that can be traded based on rarity and demand.",
        img: "https://upload.wikimedia.org/wikipedia/commons/thumb/2/2e/NFT_example.png/640px-NFT_example.png"
    },

        text: "Always manage risk: use stop-loss, take-profit, diversify your portfolio, analyze volume, and trends. Professional traders study indicators such as RSI, MACD, and moving averages.",

    }

// Function to display tutorial content
const output = document.getElementById('output');
let currentStep = 0;

document.getElementById('startBtn').addEventListener('click', ()=>{
    currentStep = 0;
    showStep(currentStep);
});

// Function to show each step
function showStep(step){
    if(step >= tutorialContent.length) {
        output.innerHTML += "<p>End of tutorial. You can ask Mara more questions below or restart!</p>";
        return;
    }
    const content = tutorialContent[step];
    output.innerHTML = `
        <h2>${content.title}</h2>
        <p>${content.text}</p>
        <img src="${content.img}" alt="${content.title}">
        <button class="toggleBtn" id="nextBtn">Next Step</button>
    `;
    if(voiceMode) speakText(content.text);
    document.getElementById('nextBtn').addEventListener('click', ()=>{
        currentStep++;
        showStep(currentStep);
    });
}

// Text-to-Speech function
function speakText(text){
    if(!('speechSynthesis' in window)) return;
    const utterance = new SpeechSynthesisUtterance(text);
    utterance.lang = 'en-US';
    speechSynthesis.speak(utterance);
}

// Infinite live updates simulation (can be replaced with live data API)
setInterval(()=>{
    if(currentStep >= tutorialContent.length) return;
    const randomTip = ["Remember to check charts daily.", "Diversify your assets for lower risk.", "Always set stop-loss orders.", "Research NFTs before buying."];
    const tip = randomTip[Math.floor(Math.random()*randomTip.length)];
    output.innerHTML += `<p><em>Live Tip:</em> ${tip}</p>`;
    output.scrollTop = output.scrollHeight;
}, 15000);
