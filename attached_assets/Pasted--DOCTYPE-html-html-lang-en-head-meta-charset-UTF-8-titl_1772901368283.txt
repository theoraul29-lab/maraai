<script>
// Mara AI activation
let maraActive = false;
document.getElementById('maraBtn').addEventListener('click', ()=>{
    maraActive = true;
    document.getElementById('output').innerHTML = "<p>Mara is now active! Ask questions or explore the cards!</p>";
});

// Reels feed demo
const reelsContainer = document.getElementById('reelsContainer');
for(let i=1;i<=7;i++){
    const div = document.createElement('div');
    div.className='reel';
    div.innerText = 'Reel #' + i;
    reelsContainer.appendChild(div);
}

// Auto highlight active reel on scroll
reelsContainer.addEventListener('scroll', ()=>{
    const reels = document.querySelectorAll('.reel');
    reels.forEach(r=>r.classList.remove('active'));
    let middle = reelsContainer.scrollTop + reelsContainer.clientHeight/2;
    reels.forEach(r=>{
        let offset = r.offsetTop + r.offsetHeight/2;
        if(Math.abs(offset - middle) < r.offsetHeight/2){
            r.classList.add('active');
        }
    });
});

// Card clicks
document.getElementById('tradingCard').addEventListener('click', ()=> alert('Open Trading Academy with charts & tutorials!'));
document.getElementById('creatorCard').addEventListener('click', ()=> alert('Open Creator Dashboard!'));
document.getElementById('funCard').addEventListener('click', ()=> alert('Open Fun Zone!'));
document.getElementById('premiumCard').addEventListener('click', ()=> alert('Show Premium Features & subscription options!'));

// Input response
const maraInput = document.getElementById('maraInput');
maraInput.addEventListener('keypress', (e)=>{
    if(e.key==='Enter' && maraActive){
        const question = maraInput.value;
        maraInput.value='';
        const output = document.getElementById('output');
        output.innerHTML += `<p><strong>You asked:</strong> ${question}</p>`;
        output.innerHTML += `<p><strong>Mara answers:</strong> Here's guidance from a 30-year experienced trader: Always analyze charts and manage risk carefully. For NFTs or DeFi, study rarity, contracts, and demand before investing.</p>`;
        output.scrollTop = output.scrollHeight;
    }
});
