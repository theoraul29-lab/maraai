<script>
// Storage for pages (demo)
let writerPages = [];

document.getElementById('createPageBtn').addEventListener('click', ()=>{

    const output = document.getElementById('output');
    output.innerHTML = `
        <h2>Create Your Writer Page</h2>
        <input type="text" id="writerName" placeholder="Your pen name">
        <input type="text" id="pageTitle" placeholder="Page Title / Book Title">
        <textarea id="contentArea" rows="8" placeholder="Write your story, poem, or chapter here..."></textarea>
        <button id="previewBtn">Preview My Page</button>
    `;

    document.getElementById('previewBtn').addEventListener('click', ()=>{
        const name = document.getElementById('writerName').value || "Anonymous";
        const title = document.getElementById('pageTitle').value || "Untitled";
        const content = document.getElementById('contentArea').value || "(No content)";
        
        // Mara AI can add suggestions (simple example)
        const suggestedEdit = "Tip from Mara AI: Add a captivating first sentence to engage readers!";
        
        // Save page
        writerPages.push({name, title, content});

        // Display preview
        output.innerHTML = `
            <h2>${title}</h2>
            <h4>by ${name}</h4>
            <p>${content}</p>
            <p style="color:#0ff;"><em>${suggestedEdit}</em></p>
            <button id="editBtn">Edit</button>
            <button id="publishBtn">Publish Online</button>
        `;

        document.getElementById('editBtn').addEventListener('click', ()=>{
            document.getElementById('createPageBtn').click();
        });

        document.getElementById('publishBtn').addEventListener('click', ()=>{
            alert(`Congratulations ${name}! Your page "${title}" is now published online with Mara AI support.`);
            // In real implementation, here we would push to a database or hosting platform
        });
    });
});
