// server.js
const express = require('express');
const bodyParser = require('body-parser');
const { Configuration, OpenAIApi } = require('openai');

const app = express();
app.use(bodyParser.json());

const configuration = new Configuration({
  apiKey: process.env.OPENAI_API_KEY // Cheia ta OpenAI
});
const openai = new OpenAIApi(configuration);

// Istoric conversații pe modul și utilizator
const conversations = {
  trading: {},
  writers: {},
  reels: {}
};

// Setări demo vs premium
const FREE_QUESTION_LIMIT = 5;

// Functie generala pentru chat
async function generateAIResponse(userId, message, module, isPremium, voice = "default") {
  if (!conversations[module][userId]) conversations[module][userId] = [];

  // Verificare demo limit
  if (!isPremium && conversations[module][userId].filter(m => m.role === 'user').length >= FREE_QUESTION_LIMIT) {
    return "Upgrade to premium to continue asking questions!";
  }

  // Adauga mesajul utilizatorului in istoric
  conversations[module][userId].push({ role: 'user', content: message });

  // Setare prompt specializat pe modul
  let systemPrompt = "";
  switch(module) {
    case "trading":
      systemPrompt = "You are Mara AI, a professional trading mentor. Explain spot, futures, NFT, DeFi, and beginner strategies. Use clear, beginner-friendly language for free users, and full detailed guidance for premium users.";
      break;
    case "writers":
      systemPrompt = "You are Mara AI, a creative writing mentor. Help users write stories, poems, and articles. Provide feedback, title suggestions, and style improvements.";
      break;
    case "reels":
      systemPrompt = "You are Mara AI, a short-video learning assistant. Explain the content of each video, summarize lessons, and guide users interactively.";
      break;
    default:
      systemPrompt = "You are Mara AI, an intelligent mentor.";
  }

  const completion = await openai.createChatCompletion({
    model: "gpt-4",
    messages: [
      { role: "system", content: systemPrompt },
      ...conversations[module][userId]
    ],
    max_tokens: 800,
    temperature: 0.7
  });

  const reply = completion.data.choices[0].message.content;
  conversations[module][userId].push({ role: 'assistant', content: reply });

  // Returneaza text + info voce
  return { reply, voice };
}

// Endpoints pentru fiecare modul

app.post('/chat/trading', async (req, res) => {
  const { userId, message, isPremium, voice } = req.body;
  const result = await generateAIResponse(userId, message, "trading", isPremium, voice);
  res.json(result);
});

app.post('/chat/writers', async (req, res) => {
  const { userId, message, isPremium, voice } = req.body;
  const result = await generateAIResponse(userId, message, "writers", isPremium, voice);
  res.json(result);
});

app.post('/chat/reels', async (req, res) => {
  const { userId, message, isPremium, voice } = req.body;
  const result = await generateAIResponse(userId, message, "reels", isPremium, voice);
  res.json(result);
});

// Pornire server
app.listen(5000, () => console.log("Mara AI multi-modul running on port 5000"));
