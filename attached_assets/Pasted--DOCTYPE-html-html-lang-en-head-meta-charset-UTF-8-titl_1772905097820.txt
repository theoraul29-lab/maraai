<script>
// Subscription system
let isSubscribed = false;
let subscriptionType = null;

// Storage for reels (demo, can be replaced with DB/API)
let reels = [
  {src:"https://www.w3schools.com/html/mov_bbb.mp4", title:"Reel 1", creator:"Alice"},
  {src:"https://www.w3schools.com/html/movie.mp4", title:"Reel 2", creator:"Bob"}
];

// Subscribe
function subscribe(type){
    subscriptionType = type;
    isSubscribed = true;
    alert(`Subscription activated: ${type==='monthly'?'1 month':'1 year'} full access!`);
    loadFeed();
}

// Add new reel (creator)
function addReel(){
    const url = prompt("Enter video URL:");
    const title = prompt("Enter video title:");
    const creator = prompt("Creator name:");
    if(url && title && creator){
        if(!isSubscribed && reels.filter(r=>r.creator===creator).length>=3){
            return alert("Free account limit: max 3 reels per month. Subscribe for unlimited posts.");
        }
        reels.push({src:url, title:title, creator:creator});
        loadFeed();
    }
}

// Load feed
function loadFeed(){
    const feed = document.getElementById('feed');
    feed.innerHTML = '';
    reels.forEach((r,index)=>{
        const div = document.createElement('div');
        div.className = 'reel';
        div.innerHTML = `
            <video src="${r.src}" muted playsinline></video>
            <button onclick="openFullscreen(${index})">Fullscreen + Sound</button>
            <div class="commentBox">
                <input type="text" placeholder="Comment..." onkeydown="if(event.key==='Enter') addComment(${index},this.value); this.value=''">
            </div>
        `;
        feed.appendChild(div);
    });
    setupScrollAutoplay();
}

// Autoplay on scroll
function setupScrollAutoplay(){
    const feed = document.getElementById('feed');
    const videos = document.querySelectorAll('video');
    feed.addEventListener('scroll', ()=>{
        videos.forEach(v=>{
            const rect = v.getBoundingClientRect();
            if(rect.top>=0 && rect.bottom<=window.innerHeight){
                v.play();
            } else {
                v.pause();
            }
        });
    });
}

// Fullscreen + sound
function openFullscreen(index){
    const video = document.querySelectorAll('video')[index];
    if(video.requestFullscreen) video.requestFullscreen();
    video.muted=false;
    video.play();
    alert("Swipe down or ESC to exit fullscreen.");
}

// Add comment (demo)
function addComment(index, text){
    if(!text) return;
    alert(`Mara AI notes: "${text}" added to reel #${index+1}`);
    // Here you can integrate Mara AI feedback system
}

// Initial load
loadFeed();
</script>
