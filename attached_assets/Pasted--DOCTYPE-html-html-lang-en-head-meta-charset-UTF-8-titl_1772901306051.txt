<script>
// Mara activation
let maraActive = false;
document.getElementById('maraBtn').addEventListener('click', ()=>{
    maraActive = true;
    document.getElementById('output').innerHTML = "<p>Mara is now active! Ask questions or explore the cards!</p>";
});

// Example fun / reels feed
const reelsContainer = document.getElementById('reelsContainer');

    div.className='reel';
    div.innerText = 'Reel #' + i;
    reelsContainer.appendChild(div);
}

// Example input response
const maraInput = document.getElementById('maraInput');
maraInput.addEventListener('keypress', (e)=>{
    if(e.key==='Enter' && maraActive){
        const question = maraInput.value;
        maraInput.value='';
        const output = document.getElementById('output');
        output.innerHTML += `<p><strong>You asked:</strong> ${question}</p>`;
        output.innerHTML += `<p><strong>Mara answers:</strong> Here’s a fun fact or tip from a 30-year experienced trader: Always check the chart trend before investing!</p>`;
        output.scrollTop = output.scrollHeight;
    }
});
