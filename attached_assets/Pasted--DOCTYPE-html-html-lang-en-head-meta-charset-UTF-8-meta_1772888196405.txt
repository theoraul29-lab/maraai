    // Connect to Mara backend WebSocket
    const ws = new WebSocket('ws://localhost:3000');

    const chatMessages = document.getElementById('chatMessages');
    const messageInput = document.getElementById('messageInput');
    const sendBtn = document.getElementById('sendBtn');
    const feed = document.getElementById('feed');

    ws.onopen = () => {
        console.log('Connected to Mara backend');
    };

    ws.onmessage = (event) => {
        const data = JSON.parse(event.data);

        // --- Update theme based on personality ---
        document.body.style.backgroundColor = data.personality.bg;
        document.body.style.color = data.personality.accent;

        // --- Display video feed ---
        feed.innerHTML = '';
        data.feedVideo.forEach(v => {
            const card = document.createElement('div');
            card.className = 'videoCard';
            const video = document.createElement('video');
            video.src = v.url;
            video.controls = true;
            card.appendChild(video);
            feed.appendChild(card);
        });

        // --- Display AI message ---
        const aiMsg = document.createElement('div');
        aiMsg.textContent = data.AIContent.text;
        aiMsg.style.margin = '5px 0';
        aiMsg.style.fontWeight = 'bold';
        chatMessages.appendChild(aiMsg);
        chatMessages.scrollTop = chatMessages.scrollHeight;
    };

    // Send message on button click or Enter key
    function sendMessage() {
        const msg = messageInput.value.trim();
        if(msg) {
            ws.send(JSON.stringify({ type: 'chat', value: msg }));
            const userMsg = document.createElement('div');
            userMsg.textContent = `You: ${msg}`;
            userMsg.style.margin = '5px 0';
            userMsg.style.color = '#333';
            chatMessages.appendChild(userMsg);
            chatMessages.scrollTop = chatMessages.scrollHeight;
            messageInput.value = '';
        }
    }
    sendBtn.addEventListener('click', sendMessage);
    messageInput.addEventListener('keydown', (e) => {
        if(e.key === 'Enter') sendMessage();
    });

    // Send scroll event as demo for personality detection
    window.addEventListener('scroll', () => {
        ws.send(JSON.stringify({ type: 'scroll', value: window.scrollY }));
    });

<script>
// --- Memorie Mara 24h ---
let userInteractions = []; // memoreaza fiecare interactiune user

// Functie pentru a adauga interactiuni
function addInteraction(type, data) {
    const timestamp = new Date().getTime();
    userInteractions.push({ type, data, timestamp });
}

// Functie pentru a curata memoria mai veche de 24h
function cleanOldMemory() {
    const now = new Date().getTime();
    const cutoff = now - (24 * 60 * 60 * 1000); // 24 ore in ms
    userInteractions = userInteractions.filter(i => i.timestamp > cutoff);
}

// Functie pentru analiza personalitate pe baza memoriei actualizate
function analyzePersonalityFromMemory() {
    cleanOldMemory(); // curata memoria veche
    // aici se face analiza in functie de interactiuni
    // demo simplu: calculam "statistici"
    let scrollSpeed = 0;
    let timeOnVideo = 0;
    let chatToneCounts = { friendly:0, aggressive:0, neutral:0 };

    userInteractions.forEach(i => {
        if(i.type === 'scroll') scrollSpeed += i.data;
        if(i.type === 'videoTime') timeOnVideo += i.data;
        if(i.type === 'chatTone') chatToneCounts[i.data] = (chatToneCounts[i.data] || 0) + 1;
    });

    // Determinam dominant personality
    let dominantTone = Object.keys(chatToneCounts).reduce((a,b)=> chatToneCounts[a] > chatToneCounts[b]?a:b);
    return detectBehavior({ scrollSpeed, timeOnVideo, chatTone: dominantTone });
}

// --- Modificam intervalul de update tema ---
setInterval(() => {
    // Adaugam interactiuni random demo (trebuie legat la backend Mara real)
    addInteraction('scroll', calculateScrollSpeed());
    addInteraction('videoTime', getTimeOnCurrentVideo());
    addInteraction('chatTone', analyzeChatTone());

    const personality = analyzePersonalityFromMemory();
    updateTheme(personality);
}, 5000);

</script>

async function searchWeb(query){

let results = [];

for(const engine of engines){

try{

const res = await axios.get(engine + encodeURIComponent(query));

results.push(res.data);

}catch(err){

console.log("search error");

}

}

return results;

}
